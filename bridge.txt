- name: Upload Artifact
        id: upload-veracode
        uses: actions/upload-artifact@v4
        with:
          name: ${{ matrix.image_name }}-veracode-scan
          path: ${{ matrix.image_name }}-veracode.zip

      - name: Generate Step Summary
        run: |
          # Grabbing the exact bytes to match your logs
          FILE_SIZE_BYTES=$(stat -c%s "${{ matrix.image_name }}-veracode.zip")
          
          echo "### Artifacts" >> $GITHUB_STEP_SUMMARY
          echo "Produced during runtime" >> $GITHUB_STEP_SUMMARY
          echo "" >> $GITHUB_STEP_SUMMARY
          echo "| Name | Size | Digest | Link |" >> $GITHUB_STEP_SUMMARY
          echo "| :--- | :--- | :--- | :--- |" >> $GITHUB_STEP_SUMMARY
          echo "| 📦 ${{ matrix.image_name }}-veracode-scan | $FILE_SIZE_BYTES Bytes | ${{ steps.upload-veracode.outputs.artifact-digest }} | [Download Zip](${{ steps.upload-veracode.outputs.artifact-url }}) |" >> $GITHUB_STEP_SUMMARY
