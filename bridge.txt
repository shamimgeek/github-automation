import time
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

TARGET_URL = "https://waltz.intranet.db.com/compositionviewer/155780-1/artifacts"

def debug_dropdown_dom(url):
    driver = webdriver.Edge()
    try:
        driver.get(url)
        print("Waiting for page load...")
        WebDriverWait(driver, 20).until(EC.presence_of_element_located((By.CLASS_NAME, "v-data-footer__select")))
        time.sleep(2)
        
        # Click the dropdown trigger
        dropdown_trigger = driver.find_element(By.CSS_SELECTOR, ".v-data-footer__select .v-input__control, .v-data-footer__select .v-select__selections")
        driver.execute_script("arguments[0].click();", dropdown_trigger)
        time.sleep(2)
        
        print("\n=== DOM OVERLAY INVESTIGATION ===")
        # Look for standard Vuetify menu boxes or active overlays
        overlays = driver.find_elements(By.CSS_SELECTOR, ".v-menu__content, .v-overlay-container, [role='listbox']")
        
        if not overlays:
            print("No explicit menu overlays caught by class. Scraping trailing root body elements...")
            overlays = driver.find_elements(By.XPATH, "/html/body/div[last()]")

        for idx, overlay in enumerate(overlays):
            print(f"\nOverlay Index [{idx}] -> Tag: {overlay.tag_name} | Class: {overlay.get_attribute('class')} | ID: {overlay.get_attribute('id')}")
            print("Inner HTML Layout snippet:")
            print(overlay.get_attribute('innerHTML')[:1500]) # Print first 1500 chars of structure
        print("=================================\n")
                
    except Exception as e:
        print(f"Diagnostic failed: {e}")
    finally:
        driver.quit()

if __name__ == "__main__":
    debug_dropdown_dom(TARGET_URL)
