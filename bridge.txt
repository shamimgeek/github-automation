import re
import time
import pandas as pd
from bs4 import BeautifulSoup
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

# Setup Configuration
TARGET_URL = "https://your-internal-dashboard-url.com/path"  # Replace with your actual URL

def scrape_all_dashboard_data(url):
    options = webdriver.EdgeOptions()
    options.add_argument('--start-maximized')
    # options.add_argument('--headless') # Keep commented to monitor execution visually
    
    driver = webdriver.Edge(options=options)
    parsed_data = []

    try:
        print("Navigating to target page...")
        driver.get(url)
        
        # 1. Wait for the primary layout structure to settle
        print("Waiting for the table container to load...")
        WebDriverWait(driver, 20).until(
            EC.presence_of_element_located((By.CLASS_NAME, "v-data-footer__select"))
        )
        time.sleep(2) 

        # 2. Click the 'Rows per page' dropdown menu trigger (Vuetify structure)
        print("Clicking 'Rows per page' dropdown selector...")
        dropdown_trigger = WebDriverWait(driver, 15).until(
            EC.element_to_be_clickable((By.CSS_SELECTOR, ".v-data-footer__select .v-input__control, .v-data-footer__select .v-select__selections"))
        )
        # Using JS click to bypass Vuetify layout element overlapping rules
        driver.execute_script("arguments[0].click();", dropdown_trigger)
        time.sleep(1.5)

        # 3. Select 'All' from the Vuetify dynamic menu list popup
        print("Selecting 'All' to expand row count...")
        all_option = WebDriverWait(driver, 15).until(
            EC.element_to_be_clickable((By.XPATH, "//div[contains(@class, 'v-list-item__title') and text()='All'] | //div[contains(@class, 'v-list-item') and .//text()='All'] | //*[text()='All']"))
        )
        driver.execute_script("arguments[0].click();", all_option)
        
        # 4. Wait for the page to completely reload all rows (e.g. 134 items)
        print("Waiting for data table to completely expand...")
        time.sleep(6) 

        # 5. Extract expanded page source and parse with BeautifulSoup
        soup = BeautifulSoup(driver.page_source, 'html.parser')
        
        # Vuetify tables usually group rows inside 'tr' elements or custom flex containers
        rows = soup.find_all(['tr', 'div'], class_=re.compile(r'(row|table-row|grid-row)', re.I)) or soup.find_all('tr')
        
        print(f"Total rows rendered on screen: {len(rows)}. Extracting valid target entries...")

        for row in rows:
            text_content = row.get_text(separator="||")
            
            # Identify row signature match
            if "com/db/pr2" in text_content:
                # Target path components, Dates, and Lifecycle options
                match_path = re.search(r'com/db/pr2/([\w\-]+)/v?([\d\.]+)/manifest\.json', text_content)
                match_date = re.search(r'(\d{2}-[A-Za-z]{3}-\d{4})', text_content)
                match_lifecycle = re.search(r'(RELEASE_CANDIDATE|PROD_LIVE|RELEASE|SNAPSHOT|DEVELOPMENT)', text_content)
                
                if match_path and match_date and match_lifecycle:
                    parsed_data.append({
                        "Binary Name": match_path.group(1),
                        "Version": match_path.group(2),
                        "Artifact Modified Date": match_date.group(1),
                        "Binary Lifecycle": match_lifecycle.group(1)
                    })

    except Exception as e:
        print(f"\nAn error occurred during execution: {e}")
        
    finally:
        driver.quit()

    return parsed_data

# --- Execution ---
if __name__ == "__main__":
    extracted_list = scrape_all_dashboard_data(TARGET_URL)
    
    if extracted_list:
        df = pd.DataFrame(extracted_list)
        # Drop duplicates if multiple interior element tree tags were picked up
        df = df.drop_duplicates()
        
        print(f"\nSuccessfully Grabbed {len(df)} Unique Rows via Edge:")
        print(df.to_string(index=False))
        
        # Save output
        df.to_csv("edge_all_binaries.csv", index=False)
        print("\nAll records saved cleanly to 'edge_all_binaries.csv'")
    else:
        print("\nNo matching rows could be extracted. Please check execution logs above.")
