import time
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.common.keys import Keys
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

# Setup Configuration
TARGET_URL = "https://your-internal-dashboard-url.com/path"  # Replace with your actual URL

def scrape_entire_table_data(url):
    options = webdriver.EdgeOptions()
    options.add_argument('--start-maximized')
    
    driver = webdriver.Edge(options=options)
    all_table_data = []

    try:
        print("Navigating to target page...")
        driver.get(url)
        
        # 1. Wait for the page structure to settle
        print("Waiting for the footer container to load...")
        WebDriverWait(driver, 20).until(
            EC.presence_of_element_located((By.CLASS_NAME, "v-data-footer__select"))
        )
        time.sleep(2) 

        # 2. Force open the dropdown using keyboard actions on the hidden input element
        print("Targeting dropdown input and forcing menu open...")
        # Locates the actual invisible input field Vuetify uses for form control
        hidden_input = WebDriverWait(driver, 15).until(
            EC.presence_of_element_located((By.CSS_SELECTOR, ".v-data-footer__select input"))
        )
        
        # Clicking the input directly and passing space/down key forces the Vuetify menu overlay visible
        hidden_input.click()
        time.sleep(0.5)
        hidden_input.send_keys(Keys.SPACE) 
        time.sleep(1.5)

        # 3. Find and select 'All' option from the newly visible floating list
        print("Searching for and selecting 'All' option...")
        all_option = WebDriverWait(driver, 15).until(
            EC.element_to_be_clickable((By.XPATH, "//div[contains(@class, 'v-list-item__title') and text()='All'] | //div[contains(@class, 'v-list-item') and .//text()='All'] | //*[text()='All']"))
        )
        driver.execute_script("arguments[0].click();", all_option)
        
        # 4. Wait for the rows to dynamically build in the browser window
        print("Waiting for table to fully update layout with all records...")
        time.sleep(6) 

        # 5. Live extraction of all rendered rows inside the active table body 
        print("Locating all populated rows...")
        rows = driver.find_elements(By.XPATH, "//table//tbody//tr | //div[@role='rowgroup']//div[@role='row']")
        
        # Backup selector fallback if standard layout tags are omitted
        if not rows:
            rows = driver.find_elements(By.CSS_SELECTOR, "tr")

        print(f"Extracting and storing data from all {len(rows)} matching table rows...")

        for row in rows:
            # Grabbing child text cell configurations
            cells = row.find_elements(By.XPATH, "./td | ./div[@role='cell']")
            row_data = [cell.text.strip().replace("\n", " ") for cell in cells if cell.text.strip()]
            
            # Save if the parsed matrix list isn't empty
            if row_data:
                all_table_data.append(row_data)

    except Exception as e:
        print(f"\nAn error occurred during execution: {e}")
        
    finally:
        driver.quit()

    return all_table_data


# --- Execution ---
if __name__ == "__main__":
    # Extracted list of lists variable
    scraped_table_variable = scrape_entire_table_data(TARGET_URL)
    
    if scraped_table_variable:
        print(f"\nSuccessfully stored {len(scraped_table_variable)} rows inside Python variable!")
        print("-" * 80)
        
        # Print a preview of the records cleanly
        print(f"Total Rows Extracted: {len(scraped_table_variable)}")
        print("Previewing First 3 Rows:")
        for index, row in enumerate(scraped_table_variable[:3]):
            print(f"Row {index + 1}: {row}")
            
        print("-" * 80)
        print(f"Python Variable Datatype: {type(scraped_table_variable)}")
    else:
        print("\nFailed to expand the table dynamically. Variable is empty.")
