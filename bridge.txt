import time
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

TARGET_URL = "https://your-internal-dashboard-url.com/path" # Replace with your actual URL

def debug_dropdown_options(url):
    driver = webdriver.Edge()
    try:
        driver.get(url)
        print("Waiting for page load...")
        WebDriverWait(driver, 20).until(EC.presence_of_element_located((By.CLASS_NAME, "v-data-footer__select")))
        
        # Open the dropdown wrapper
        dropdown_trigger = driver.find_element(By.CSS_SELECTOR, ".v-data-footer__select .v-input__control, .v-data-footer__select .v-select__selections")
        driver.execute_script("arguments[0].click();", dropdown_trigger)
        time.sleep(2)
        
        print("\n--- DIAGNOSTIC DATA ---")
        # Find every single visible overlay menu list item on screen
        menu_items = driver.find_elements(By.XPATH, "//div[contains(@class, 'v-list-item')] | //*[contains(@id, 'list-item')]")
        
        if not menu_items:
            print("No items found using primary lookups. Scraping all text inside active overlay containers...")
            menu_items = driver.find_elements(By.CSS_SELECTOR, ".v-menu__content, .v-overlay-container")
            
        for idx, item in enumerate(menu_items):
            text = item.text.strip().replace('\n', ', ')
            if text:
                print(f"Option index [{idx}]: Text value -> '{text}'")
        print("-----------------------\n")
                
    except Exception as e:
        print(f"Diagnostic failed: {e}")
    finally:
        driver.quit()

if __name__ == "__main__":
    debug_dropdown_options(TARGET_URL)
