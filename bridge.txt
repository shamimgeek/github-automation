import time
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.common.action_chains import ActionChains
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

# Setup Configuration
TARGET_URL = "https://waltz.intranet.db.com/compositionviewer/155780-1/artifacts" 

def scrape_entire_table_data(url):
    options = webdriver.EdgeOptions()
    options.add_argument('--start-maximized')
    
    driver = webdriver.Edge(options=options)
    all_table_data = []

    try:
        print("Navigating to target page...")
        driver.get(url)
        
        # 1. Wait for the pagination footer section layout to render completely
        print("Waiting for the footer container to load...")
        WebDriverWait(driver, 20).until(
            EC.presence_of_element_located((By.CLASS_NAME, "v-data-footer__select"))
        )
        time.sleep(3) 

        # 2. Trigger the 'Rows per page' dropdown menu to pop open
        print("Clicking 'Rows per page' dropdown selector...")
        dropdown_trigger = WebDriverWait(driver, 15).until(
            EC.element_to_be_clickable((By.CSS_SELECTOR, (
                ".v-data-footer__select .v-select__slot, "
                ".v-data-footer__select .v-select__selections, "
                ".v-data-footer__select .v-input__control"
            )))
        )
        
        try:
            dropdown_trigger.click()
        except Exception:
            try:
                ActionChains(driver).move_to_element(dropdown_trigger).click().perform()
            except Exception:
                driver.execute_script("arguments[0].click();", dropdown_trigger)
                
        print("Waiting for dropdown options menu overlay to pop up...")
        time.sleep(2.5)

        # 3. Target 'All' strictly inside Vuetify's active menu box popover
        print("Selecting 'All' from the active menu overlay...")
        all_option = WebDriverWait(driver, 15).until(
            EC.element_to_be_clickable((By.XPATH, (
                "//div[contains(@class, 'menuable__content__active')]//*[text()='All' or normalize-space()='All']"
            )))
        )
        driver.execute_script("arguments[0].click();", all_option)
        
        # 4. Give the browser ample time to rebuild the table grid with all 134 entries
        print("Waiting for data table layout to fully expand records...")
        time.sleep(7) 

        # 5. Grab all rows currently living in the expanded UI matrix
        print("Locating all live data rows on screen...")
        rows = driver.find_elements(By.XPATH, "//table//tbody//tr | //div[@role='rowgroup']//div[@role='row']")
        
        if not rows:
            rows = driver.find_elements(By.CSS_SELECTOR, "tr")

        print(f"Extracting cells and writing to variable from all {len(rows)} rows...")

        for row in rows:
            cells = row.find_elements(By.XPATH, "./td | ./div[@role='cell']")
            row_data = [cell.text.strip().replace("\n", " ") for cell in cells if cell.text.strip()]
            
            # Ensure the row is valid and has at least the path column
            if row_data and len(row_data) > 0:
                full_path = row_data[0]
                image_name = "N/A"
                version = "N/A"
                
                # Dynamic Parsing Logic for the path string
                if "manifest.json" in full_path:
                    # Example split: ['com', 'db', 'pr2', 'rclone', 'v1.6.0', 'manifest.json']
                    parts = full_path.split('/')
                    if len(parts) >= 3:
                        # Version sits right before manifest.json, Image Name sits right before version
                        version = parts[-2]
                        image_name = parts[-3]
                
                # Append a clean structured dictionary to your list variable
                all_table_data.append({
                    "image_name": image_name,
                    "version": version,
                    "full_path": full_path,
                    "raw_row_data": row_data  # Keeps the rest of the columns intact
                })

    except Exception as e:
        print(f"\nAn error occurred during execution: {e}")
        
    finally:
        driver.quit()

    return all_table_data


# --- Execution ---
if __name__ == "__main__":
    scraped_table_variable = scrape_entire_table_data(TARGET_URL)
    
    if scraped_table_variable:
        print(f"\nSuccessfully stored {len(scraped_table_variable)} rows inside Python variable!")
        print("-" * 120)
        print(f"{'IMAGE NAME':<25} | {'VERSION':<15} | {'FULL PATH'}")
        print("-" * 120)
        
        # Preview the parsing results for the first 10 rows
        for item in scraped_table_variable[:10]:
            print(f"{item['image_name']:<25} | {item['version']:<15} | {item['full_path']}")
            
        print("-" * 120)
        print("\nStructure Sample of a single entry inside the variable:")
        print(scraped_table_variable[1])
